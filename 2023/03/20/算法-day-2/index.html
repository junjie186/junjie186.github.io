有序数组的平方

给你一个按 非递减顺序 排序的整数数组 nums，返回 每个数字的平方 组成的新数组，要求也按 非递减顺序 排序。

示例 1： 输入：nums = [-4,-1,0,3,10] 输出：[0,1,9,16,100] 解释：平方后，数组变为 [16,1,0,9,100]，排序后，数组变为 [0,1,9,16,100]

示例 2： 输入：nums = [-7,-3,2,3,11] 输出：[4,9,9,49,121]

我的思路:找到数组的正负分界,然后用归并排序.

使用双指针法更简单

提交:

class Solution {
public:
    vector<int> sortedSquares(vector<int>& nums) {
        int index1=0;
        while(index1!=nums.size()){
            if(nums[index1]>0){
                break;
            }
            index1++;
        }
        int index2=index1-1;
        //没有初始化长度
        vector<int> ret;
        while(index2!=-1&&index1!=nums.size()){
            if(nums[index1]*nums[index1]<=nums[index2]*nums[index2]){
                ret.push_back(nums[index1]*nums[index1]);
                index1++;
            }else{
                ret.push_back(nums[index2]*nums[index2]);
                index2--;
            }
        }
        for(;index1<nums.size();index1++){
            ret.push_back(nums[index1]*nums[index1]);
        }
        for(;index2>=0;index2--){
            ret.push_back(nums[index2]*nums[index2]);
        }
        return ret;
    }
};

例子,使用双指针法:

class Solution {
public:
    vector<int> sortedSquares(vector<int>& A) {
        int k = A.size() - 1;
        vector<int> result(A.size(), 0);
        for (int i = 0, j = A.size() - 1; i <= j;) { // 注意这里要i <= j，因为最后要处理两个元素
            if (A[i] * A[i] < A[j] * A[j])  {
                result[k--] = A[j] * A[j];
                j--;
            }
            else {
                result[k--] = A[i] * A[i];
                i++;
            }
        }
        return result;
    }
};

长度最小的子数组

给定一个含有 n 个正整数的数组和一个正整数 s ，找出该数组中满足其和 ≥ s 的长度最小的 连续 子数组，并返回其长度。如果不存在符合条件的子数组，返回 0。

示例：

输入：s = 7, nums = [2,3,1,2,4,3] 输出：2 解释：子数组 [4,3] 是该条件下的长度最小的子数组。

思路:滑动窗口

提交:

class Solution {
public:
    int minSubArrayLen(int target, vector<int>& nums) {
        int left=0,right =0;
        int length=nums.size()+1;
        int sum=0;
        for(;right<nums.size();right++){
            sum+=nums[right];
            if(sum>=target){
                length=(right-left+1)<length?(right-left+1):length;
                for(;left<right;){
                    sum-=nums[left];
                    left++;
                    if(sum>=target){
                        length=(right-left+1)<length?(right-left+1):length;
                    }else{
                        break;
                    }
                }
            }
            
        }
        if(length>nums.size()){
            return 0;
        }else{
            return length;
        }
    }
};

例子:

class Solution {
public:
    int minSubArrayLen(int s, vector<int>& nums) {
        int result = INT32_MAX;
        int sum = 0; // 滑动窗口数值之和
        int i = 0; // 滑动窗口起始位置
        int subLength = 0; // 滑动窗口的长度
        for (int j = 0; j < nums.size(); j++) {
            sum += nums[j];
            // 注意这里使用while，每次更新 i（起始位置），并不断比较子序列是否符合条件
            while (sum >= s) {
                subLength = (j - i + 1); // 取子序列的长度
                result = result < subLength ? result : subLength;
                sum -= nums[i++]; // 这里体现出滑动窗口的精髓之处，不断变更i（子序列的起始位置）
            }
        }
        // 如果result没有被赋值的话，就返回0，说明没有符合条件的子序列
        return result == INT32_MAX ? 0 : result;
    }
};

螺旋矩阵II

给定一个正整数 n，生成一个包含 1 到 n^2 所有元素，且元素按顺时针顺序螺旋排列的正方形矩阵。

示例:

输入: 3 输出: [ [ 1, 2, 3 ], [ 8, 9, 4 ], [ 7, 6, 5 ] ]

思路:要利用循环层数

提交:

class Solution {
public:
    vector<vector<int>> generateMatrix(int n)
{
    int i = 0, j = 0;
    int num = 1;
    vector<vector<int>> res(n, vector<int>(n, 0));
    int k = 1;
    res[0][0] = k;
    k++;
    i++;
    while (k <= n * n)
    {
        switch (num % 4)
        {
        case 1:
            for (; i < n - num / 4; i++)
            {
                res[j][i] = k;
                k++;
            }
            //这里第一次提交没有写,导致出错
            i--;
            j++;
            num++;
            break;
        case 2:
            for (; j < n - num / 4; j++)
            {
                res[j][i] = k;
                k++;
            }
            j--;
            i--;
            num++;
            break;
        case 3:
            for (; i >= num / 4; i--)
            {
                res[j][i] = k;
                k++;
            }
            i++;
            j--;
            num++;
            break;
        case 0:
            for (; j >= num / 4; j--)
            {
                res[j][i] = k;
                k++;
            }
            j++;
            i++;
            num++;
            break;
        }
    }
    return res;
}

};

要尤其注意for循环后的index值,是先变化值,再进行判断
