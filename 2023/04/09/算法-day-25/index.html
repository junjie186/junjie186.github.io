组合总和 III

找出所有相加之和为 n 的 k 个数的组合，且满足下列条件：

只使用数字1到9
每个数字 最多使用一次
返回 所有可能的有效组合的列表 。该列表不能包含相同的组合两次，组合可以以任何顺序返回。

提交:

class Solution {
private:
    vector<vector<int>> result;
public:
    vector<vector<int>> combinationSum3(int k, int n) {
        vector<int> nums;
        backtracking(nums,k,n,1);
        return result;
    }

    void backtracking(vector<int>& nums,int k,int n,int start){
        if(9-start+1<k||n<(2*start+k-1)*k/2){
            return;
        }
        if(n==0&&k==0){
            result.push_back(nums);
            return;
        }
        for(int i=start;i<10;i++){
            nums.push_back(i);
            backtracking(nums,k-1,n-i,i+1);
            nums.pop_back();
        }
    }
};

这个剪枝是指在剩下的数里面不可能取到和为n的数.

下面的剪枝较为简单:

class Solution {
private:
    vector<vector<int>> result; // 存放结果集
    vector<int> path; // 符合条件的结果
    void backtracking(int targetSum, int k, int sum, int startIndex) {
        if (sum > targetSum) { // 剪枝操作
            return; // 如果path.size() == k 但sum != targetSum 直接返回
        }
        if (path.size() == k) {
            if (sum == targetSum) result.push_back(path);
            return;
        }
        for (int i = startIndex; i <= 9 - (k - path.size()) + 1; i++) { // 剪枝
            sum += i; // 处理
            path.push_back(i); // 处理
            backtracking(targetSum, k, sum, i + 1); // 注意i+1调整startIndex
            sum -= i; // 回溯
            path.pop_back(); // 回溯
        }
    }

public:
    vector<vector<int>> combinationSum3(int k, int n) {
        result.clear(); // 可以不加
        path.clear();   // 可以不加
        backtracking(n, k, 0, 1);
        return result;
    }
};

电话号码的字母组合

给定一个仅包含数字 2-9 的字符串，返回所有它能表示的字母组合。答案可以按 任意顺序 返回。

给出数字到字母的映射如下（与电话按键相同）。注意 1 不对应任何字母。

思路:回溯

提交:

class Solution {
private:
    const string letterMap[10] = {
        "", // 0
        "", // 1
        "abc", // 2
        "def", // 3
        "ghi", // 4
        "jkl", // 5
        "mno", // 6
        "pqrs", // 7
        "tuv", // 8
        "wxyz", // 9
    };

    vector<string> result;
public:
    vector<string> letterCombinations(string digits) {
        if(digits.empty()){
            return {};
        }
        string path;
        backtracking(path,digits,0);
        return result;
    }

    void backtracking(string& path,string digits,int n){
        if(n==digits.size()){
            result.push_back(path);
            return;
        }
        int num=digits[n]-'0';
        for(int i=0;i<letterMap[num].size();i++){
            path.push_back(letterMap[num][i]);
            backtracking(path,digits,n+1);
            path.pop_back();
        }
    }
};

优化:使用const string

// 版本二
class Solution {
private:
        const string letterMap[10] = {
            "", // 0
            "", // 1
            "abc", // 2
            "def", // 3
            "ghi", // 4
            "jkl", // 5
            "mno", // 6
            "pqrs", // 7
            "tuv", // 8
            "wxyz", // 9
        };
public:
    vector<string> result;
    void getCombinations(const string& digits, int index, const string& s) { // 注意参数的不同
        if (index == digits.size()) {
            result.push_back(s);
            return;
        }
        int digit = digits[index] - '0';
        string letters = letterMap[digit];
        for (int i = 0; i < letters.size(); i++) {
            getCombinations(digits, index + 1, s + letters[i]);  // 注意这里的不同
        }
    }
    vector<string> letterCombinations(string digits) {
        result.clear();
        if (digits.size() == 0) {
            return result;
        }
        getCombinations(digits, 0, "");
        return result;

    }
};
